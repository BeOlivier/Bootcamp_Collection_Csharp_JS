//////////
Create a table Customers with a Name, Email and PhoneNumber-field
//////////
    CREATE TABLE dbo.Customers (
        CustomerId int NOT NULL PRIMARY KEY,
        Name NVARCHAR(50) NOT NULL,
        Email NVARCHAR(50) NOT NULL,
        PhoneNumber NVARCHAR(50) NOT NULL
    );
    GO

//////////
Create another table Address that holds StreetName, StreetNo, City and Country and a boolean called Primary
//////////
    CREATE TABLE dbo.Address (
        AddressId int NOT NULL PRIMARY KEY,
        StreetName NVARCHAR(50) NOT NULL,
        StreetNo NVARCHAR(50) NOT NULL,
        City NVARCHAR(50) NOT NULL,
        Country NVARCHAR(50) NOT NULL,
        AddressPrimary BIT NOT NULL DEFAULT 1
    );
    GO
    
//////////
Create a foreign key from Address to Customer on the field Address.CustomerId
//////////
    ALTER TABLE dbo.Address
    ADD CustomerId int
        CONSTRAINT FK_CUSTOMER_ID
        FOREIGN KEY (CustomerId) REFERENCES Customers(CustomerId);
